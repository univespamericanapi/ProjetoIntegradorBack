Aqui listarei as rotas da api com todos os campos necessários no body ou header para efetuar uma requisição.


route: /api/auth/signup
type: post
body:
{
  "username": "",
  "password": "",
  "name": "",
  "role": ""
}
return:
{
  "message": "User was registered successfully!"
}


route: /api/auth/signin
type: post
body:
{
  "username": "",
  "password": ""
}
return:
{
  "idUser": ,
  "username": "",
  "name": "",
  "role": "",
  "accessToken": "",
  "refreshToken": ""
}


route: /api/auth/refreshtoken
type: post
body:
{
  "refreshToken": ""
}
return:
{
  "accessToken": "",
  "refreshToken": ""
}

body:
{
  "estado": ""
}

{
  "cidadeLista": []
}


route: /api/cadastro/estados
type: get
return:
{
  "estadoLista": []
}


route: /api/cadastro/categorias
type: get
return:
{
  "categoriaLista": []
}

}

  "eventoLista": []
}
